let coins = [200, 100, 50, 20, 10, 5, 2, 1]
function getChange (totalPayable, cashPaid) {
  let difference = cashPaid - totalPayable
  let result = []
  while (difference > 0) {
    let maxCoin = coins.find(v => v <= difference)
    result.push(maxCoin)
    difference -= maxCoin
  }
  return result
}
